# ArtiZone Booking & Scheduling System

Complete booking and scheduling system for ArtiZone Beauty & Aesthetic Clinic.

## Stack
- **Framework**: Next.js 14 (App Router)
- **Database**: MySQL + Drizzle ORM
- **Styling**: Tailwind CSS
- **Validation**: Zod

## Quick Start

1. **Install dependencies**
   ```bash
   npm install
   ```

2. **Set up environment variables**
   ```bash
   cp .env.example .env
   # Edit .env with your database credentials
   ```

3. **Run database migrations**
   ```bash
   npm run db:push
   ```

4. **Start development server**
   ```bash
   npm run dev
   ```

11 tables:

- `staff` + `staff_skills` + `staff_availability` + `staff_time_off` — Staff management
- `resources` — Rooms & machines

- ✅ Dual-write booking system (legacy + scheduling)
- ✅ 3-gate slot availability check (working hours, time off, overlap)
- ✅ Double-booking guard with 409 response
- ✅ Status mirroring between booking_requests and appointments
- ✅ Audit trail for all status changes
- ✅ Loyalty points on completion
- ✅ 24h reminder scheduler (cron-ready)
- ✅ WhatsApp integration
- ✅ Client self-service portal

export default async function handler(req: Request) {
  if (req.method !== "GET") return new Response("Method not allowed", { status: 405 });
  try {
    const url = new URL(req.url);
    const statusFilter = url.searchParams.get("status");
    let query = db.select().from(bookingRequests).orderBy(desc(bookingRequests.createdAt));

    return new Response(JSON.stringify({ bookings: rows }), { status: 200, headers: { "Content-Type": "application/json" } });

    return new Response(JSON.stringify({ error: "Internal server error" }), { status: 500, headers: { "Content-Type": "application/json" } });

import { db } from "~/server/db";
import { bookingRequests, appointments, appointmentEvents } from "~/server/db/schema";
import { eq } from "drizzle-orm";
import { z } from "zod";

const patchSchema = z.object({
  status: z.enum(["pending", "confirmed", "cancelled", "no_show", "declined"]).optional(),
  date: z.string().regex(/^\d{4}-\d{2}-\d{2}$/).optional(),
  time: z.string().regex(/^\d{2}:\d{2}$/).optional(),
  service: z.string().min(1).max(255).optional(),
  notes: z.string().optional(),
  adminNotes: z.string().optional(),
}).partial();

export default async function handler(req: Request, { params }: { params: { id: string } }) {
  if (req.method !== "PATCH") return new Response("Method not allowed", { status: 405 });
  try {
    const id = parseInt(params.id);
    if (isNaN(id)) return new Response("Invalid ID", { status: 400 });
    const body = await req.json();
    const parsed = patchSchema.safeParse(body);
    if (!parsed.success) return new Response(JSON.stringify({ error: "Invalid input", details: parsed.error.flatten() }), { status: 400, headers: { "Content-Type": "application/json" } });

    const updates: any = { ...parsed.data };
    if (updates.status === "confirmed") updates.confirmedAt = new Date();
    if (updates.status === "no_show") updates.noShowAt = new Date();

    await db.update(bookingRequests).set(updates).where(eq(bookingRequests.id, id));

    if (updates.status) {
      const linkedAppointments = await db.select().from(appointments).where(eq(appointments.bookingRequestId, id));
      for (const appt of linkedAppointments) {
        const apptUpdates: any = { status: updates.status };
        if (updates.status === "confirmed") apptUpdates.confirmedAt = new Date();
        if (updates.status === "no_show") apptUpdates.noShowAt = new Date();
        await db.update(appointments).set(apptUpdates).where(eq(appointments.id, appt.id));

    if (!parsed.success) return new Response(JSON.stringify({ error: "Invalid input", details: parsed.error.flatten() }), { status: 400, headers: { "Content-Type": "application/json" } });

    const { name, phone, service, date, time, notes, preferredStaffId, preferredResourceId } = parsed.data;
    const startMinutes = timeToMinutes(time);

    // Step 1: Write to booking_requests (legacy queue)
    const [bookingReq] = await db.insert(bookingRequests).values({ name, phone, service, date, time, notes: notes || null, status: "pending" }).$returningId();
    const bookingRequestId = bookingReq.id;

    // Step 2: Resolve service from catalogue (fuzzy match)
    const firstWord = service.split(/\s+/)[0];
    const serviceRows = await db.select().from(services).where(like(services.name, `%${firstWord}%`)).limit(1);
    const serviceRow = serviceRows[0];
    let durationMin = 60, bufferMin = 0, serviceId = null, category = "other";
    let defaultStaffId = null, requiredCapability = null;
    if (serviceRow) {
      durationMin = serviceRow.durationMin; bufferMin = serviceRow.bufferMin || 0;
      serviceId = serviceRow.id; category = serviceRow.category;
      defaultStaffId = serviceRow.defaultStaffId; requiredCapability = serviceRow.requiredCapability;
    }
    const endMinutes = startMinutes + durationMin + bufferMin;
    const endTime = minutesToTime(endMinutes);

    // Step 3: DOUBLE-BOOKING GUARD — 4 gates
    const allStaff = await db.select().from(staff).where(eq(staff.active, true));
    const allSkills = await db.select().from(staffSkills);
    const allAvailability = await db.select().from(staffAvailability);
    const allTimeOff = await db.select().from(staffTimeOff).where(
      and(lte(staffTimeOff.startAt, new Date(`${date}T23:59:59`)), gte(staffTimeOff.endAt, new Date(`${date}T00:00:00`)))
    );
    const existingAppointments = await db.select().from(appointments).where(
      and(eq(appointments.date, date), inArray(appointments.status, ["confirmed", "requested"]))
    );
    let availableResources = requiredCapability
      ? await db.select().from(resources).where(and(eq(resources.active, true), eq(resources.capability, requiredCapability)))
      : await db.select().from(resources).where(eq(resources.active, true));

    const qualifiedStaff = allStaff.filter(s => {
      const skills = allSkills.filter(sk => sk.staffId === s.id).map(sk => sk.skill);
      return skills.includes(category as any);
    });

    if (qualifiedStaff.length === 0) {
      return new Response(JSON.stringify({ success: true, bookingRequestId, warning: "no_qualified_staff",
        whatsappUrl: `https://wa.me/962790412758?text=Hello ArtiZone, I booked ${encodeURIComponent(service)} on ${date} at ${time} under name ${encodeURIComponent(name)}`
      }), { status: 200, headers: { "Content-Type": "application/json" } });
    }

    const weekday = new Date(date).getDay();
    let availableStaffId: number | null = null;
    let assignedResourceId: number | null = null;

    for (const s of qualifiedStaff) {
      if (preferredStaffId && s.id !== preferredStaffId) continue;
      // Gate 1: Working hours
      const avail = allAvailability.filter(a => a.staffId === s.id && a.weekday === weekday);
      const hasWorkingHours = avail.some(a => timeToMinutes(a.startTime) <= startMinutes && timeToMinutes(a.endTime) >= endMinutes);
      if (!hasWorkingHours) continue;

      // Gate 3: No overlap with existing appointments
      if (existingAppointments.some(a => a.staffId === s.id && overlaps(time, endTime, a.startTime, a.endTime))) continue;

        for (const resource of availableResources) {
          if (preferredResourceId && resource.id !== preferredResourceId) continue;
          if (!existingAppointments.some(a => a.resourceId === resource.id && overlaps(time, endTime, a.startTime, a.endTime))) {
            assignedResourceId = resource.id; roomFound = true; break;
          }
        }
        if (!roomFound) continue;
      }
      availableStaffId = s.id; break;
    }

    if (!availableStaffId) {
      return new Response(JSON.stringify({ error: "slot_unavailable",
        message: "This time slot is fully booked. Please choose another time or join the waitlist.",
        bookingRequestId, reason: availableResources.length > 0 ? "staff_or_room_unavailable" : "no_qualified_staff"
      }), { status: 409, headers: { "Content-Type": "application/json" } });
    }

      customerName: name, customerPhone: phone, serviceId, serviceName: service,
      staffId: availableStaffId, resourceId: assignedResourceId,

    for (let slotStart = timeToMinutes("09:00"); slotStart + durationMin <= timeToMinutes("21:00"); slotStart += SLOT_INTERVAL) {
      const slotEnd = slotStart + durationMin;
      const time24 = minutesToTime(slotStart);
      const timeEnd = minutesToTime(slotEnd);

      if (date === todayAmman && slotStart <= nowMinutes) {
        slots.push({ time: formatTime12(time24), time24, available: false, reason: "past" }); continue;
      }

      let isAvailable = false, reason: string | undefined = "fully_booked";
      for (const s of qualifiedStaff) {
        const avail = allAvailability.filter(a => a.staffId === s.id && a.weekday === weekday);
        if (!avail.some(a => timeToMinutes(a.startTime) <= slotStart && timeToMinutes(a.endTime) >= slotEnd)) continue;
        if (allTimeOff.some(to => to.staffId === s.id)) continue;
        if (existingAppointments.some(a => a.staffId === s.id && overlaps(time24, timeEnd, a.startTime, a.endTime))) continue;
        if (availableResources.length > 0) {
          const roomFree = availableResources.some(r =>
            !existingAppointments.some(a => a.resourceId === r.id && overlaps(time24, timeEnd, a.startTime, a.endTime))
          );
          if (!roomFree) { reason = "room_unavailable"; continue; }
        }
        isAvailable = true; break;
      }
      slots.push({ time: formatTime12(time24), time24, available: isAvailable, reason: isAvailable ? undefined : reason });
    }

    return new Response(JSON.stringify({ slots, date, durationMin }), { status: 200, headers: { "Content-Type": "application/json" } });
  } catch (error) {
    return new Response(JSON.stringify({ error: "Internal server error" }), { status: 500, headers: { "Content-Type": "application/json" } });
  }
}

function formatTime12(time24: string): string {
  const [h, m] = time24.split(":").map(Number);
  const period = h < 12 ? "AM" : "PM";
  const h12 = h === 0 ? 12 : h > 12 ? h - 12 : h;
  return `${h12}:${m.toString().padStart(2, "0")} ${period}`;
}
